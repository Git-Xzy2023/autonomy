# 泛型接口、泛型类

```ts
// 泛型接口
interface Container<T> {
  value: T;
  get(): T;
  set(v: T): void;
}
const box: Container<number> = {
  value: 42,
  get() {
    return this.value;
  },
  set(v) {
    this.value = v;
  },
};

// 泛型类
class Stack<T> {
  private items: T[] = [];
  push(item: T) {
    this.items.push(item);
  }
  pop(): T | undefined {
    return this.items.pop();
  }
}
const numStack = new Stack<number>();
numStack.push(1);
numStack.push("a"); // ❌
```
