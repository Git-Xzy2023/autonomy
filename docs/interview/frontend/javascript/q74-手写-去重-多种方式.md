# 手写 `去重` 多种方式

```js
// 1. Set
[...new Set(arr)];

// 2. filter + indexOf
arr.filter((v, i, a) => a.indexOf(v) === i);

// 3. 对象属性去重（针对对象数组按某字段）
const uniqueByKey = (arr, key) => {
  const seen = new Set();
  return arr.filter((o) =>
    seen.has(o[key]) ? false : (seen.add(o[key]), true),
  );
};
```
