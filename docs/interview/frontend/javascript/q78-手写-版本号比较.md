# 鎵嬪啓 `鐗堟湰鍙锋瘮杈僠

```js
function compareVersion(v1, v2) {
  const a = v1.split(".").map(Number);
  const b = v2.split(".").map(Number);
  const len = Math.max(a.length, b.length);
  for (let i = 0; i < len; i++) {
    const x = a[i] || 0;
    const y = b[i] || 0;
    if (x > y) return 1;
    if (x < y) return -1;
  }
  return 0;
}
compareVersion("1.2.3", "1.2.10"); // -1
compareVersion("1.20.0", "1.3.0"); // 1
```
