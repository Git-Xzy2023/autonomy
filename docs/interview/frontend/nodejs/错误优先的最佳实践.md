# 错误优先的最佳实践

```js
// ✓ 每一个回调都检查 err
fs.readFile(path, (err, data) => {
  if (err) {
    // 正确处理：向上传递 / 记录日志 / 给用户友好响应
    return callback(err);
  }
  // ...
});

// ✓ async 函数用 try/catch
async function safe() {
  try {
    await risky();
  } catch (err) {
    logger.error(err);
    throw new AppError("操作失败", 500); // 包装成业务错误
  }
}

// ✓ 区分可恢复错误（EEXIST 文件已存在）和系统错误（ENOMEM）
if (err.code === "ENOENT") {
  /* 文件不存在 */
}
if (err.code === "EADDRINUSE") {
  /* 端口被占 */
}
```
