# Vue3 渲染器原理与自定义渲染器

## 渲染器的职责

渲染器（Renderer）负责**把 VNode 转成真实 DOM**（挂载/更新/删除）。Vue3 把渲染器抽象成"平台无关"的核心，通过传入不同的"平台操作函数"，可以渲染到不同目标（浏览器 DOM、Canvas、WebGL、原生移动端）。

## 渲染器的核心 API

```js
// packages/runtime-core/src/renderer.ts
export function createRenderer(options) {
  // options：平台操作函数（DOM 操作）
  const {
    insert: hostInsert,
    remove: hostRemove,
    patchProp: hostPatchProp,
    createElement: hostCreateElement,
    createText: hostCreateText,
    createComment: hostCreateComment,
    setText: hostSetText,
    setElementText: hostSetElementText,
    parentNode: hostParentNode,
    nextSibling: hostNextSibling
  } = options;

  // 内部函数：patch, mountElement, patchElement, mountChildren, etc.
  function patch(n1, n2, container, anchor) { /* ... */ }
  function mountElement(vnode, container, anchor) { /* ... */ }
  function patchElement(n1, n2) { /* ... */ }
  function mountChildren(children, container, anchor) { /* ... */ }
  function unmount(vnode) { /* ... */ }

  const render = (vnode, container) => {
    if (vnode == null) {
      if (container._vnode) {
        unmount(container._vnode);
      }
    } else {
      patch(container._vnode || null, vnode, container);
    }
    container._vnode = vnode;
  };

  return { render, hydrate, createApp: createAppAPI(render, hydrate) };
}
```

## 浏览器渲染器

```js
// packages/runtime-dom/src/index.ts
import { createRenderer } from '@vue/runtime-core';

// DOM 操作函数
const renderer = createRenderer({
  insert(child, parent, anchor) {
    parent.insertBefore(child, anchor || null);
  },
  remove(child) {
    const parent = child.parentNode;
    if (parent) parent.removeChild(child);
  },
  createElement(tag, isSVG) {
    return isSVG ? document.createElementNS('http://www.w3.org/2000/svg', tag)
                 : document.createElement(tag);
  },
  createText(text) {
    return document.createTextNode(text);
  },
  createComment(text) {
    return document.createComment(text);
  },
  setText(node, text) {
    node.nodeValue = text;
  },
  setElementText(el, text) {
    el.textContent = text;
  },
  parentNode(node) {
    return node.parentNode;
  },
  nextSibling(node) {
    return node.nextSibling;
  },
  patchProp(el, key, prevValue, nextValue, isSVG) {
    if (key === 'class') {
      el.className = nextValue || '';
    } else if (key === 'style') {
      for (const k in nextValue) {
        el.style[k] = nextValue[k];
      }
    } else if (key.startsWith('on')) {
      const event = key.slice(2).toLowerCase();
      if (nextValue) {
        el.addEventListener(event, nextValue);
      } else {
        el.removeEventListener(event, prevValue);
      }
    } else if (key in el) {
      el[key] = nextValue;
    } else {
      if (nextValue == null) {
        el.removeAttribute(key);
      } else {
        el.setAttribute(key, nextValue);
      }
    }
  }
});

export const createApp = renderer.createApp;
```

## patch 的核心流程

```js
function patch(n1, n2, container, anchor) {
  const { type, shapeFlag } = n2;

  if (n1 && !isSameVNodeType(n1, n2)) {
    // 类型不同：卸载旧的，挂载新的
    unmount(n1);
    n1 = null;
  }

  switch (type) {
    case Text:
      processText(n1, n2, container, anchor);
      break;
    case Comment:
      processCommentNode(n1, n2, container, anchor);
      break;
    case Static:
      processStaticNode(n1, n2, container, anchor);
      break;
    case Fragment:
      processFragment(n1, n2, container, anchor);
      break;
    default:
      if (shapeFlag & ShapeFlags.ELEMENT) {
        processElement(n1, n2, container, anchor);
      } else if (shapeFlag & ShapeFlags.COMPONENT) {
        processComponent(n1, n2, container, anchor);
      } else if (shapeFlag & ShapeFlags.TELEPORT) {
        type.process(n1, n2, container, anchor, parentComponent, suspense);
      } else if (shapeFlag & ShapeFlags.SUSPENSE) {
        type.process(n1, n2, container, anchor, parentComponent, suspense);
      }
  }
}

function processElement(n1, n2, container, anchor) {
  if (n1 == null) {
    mountElement(n2, container, anchor);
  } else {
    patchElement(n1, n2);
  }
}

function mountElement(vnode, container, anchor) {
  const { type, props, children, shapeFlag } = vnode;
  const el = vnode.el = hostCreateElement(type, isSVG);

  if (props) {
    for (const key in props) {
      hostPatchProp(el, key, null, props[key]);
    }
  }

  if (shapeFlag & ShapeFlags.TEXT_CHILDREN) {
    hostSetElementText(el, children);
  } else if (shapeFlag & ShapeFlags.ARRAY_CHILDREN) {
    mountChildren(children, el, null);
  }

  hostInsert(el, container, anchor);
}
```

## 自定义渲染器

Vue3 的渲染器是平台无关的，可以渲染到任意目标。

### 示例：Canvas 渲染器

```js
import { createRenderer } from '@vue/runtime-core';

const canvas = document.getElementById('canvas');
const ctx = canvas.getContext('2d');

const renderer = createRenderer({
  insert(child, parent, anchor) {
    // Canvas 没有 DOM 结构，用自定义树
    if (!parent.children) parent.children = [];
    parent.children.push(child);
    child.parent = parent;
  },
  remove(child) {
    const parent = child.parent;
    const i = parent.children.indexOf(child);
    if (i > -1) parent.children.splice(i, 1);
  },
  createElement(tag) {
    return { tag, props: {}, children: [] };
  },
  createText(text) {
    return { text };
  },
  setElementText(el, text) {
    el.text = text;
  },
  patchProp(el, key, prev, next) {
    el.props[key] = next;
  },
  parentNode(node) {
    return node.parent;
  },
  nextSibling(node) {
    const parent = node.parent;
    if (!parent) return null;
    const i = parent.children.indexOf(node);
    return parent.children[i + 1] || null;
  },
  // ...
});

// 自定义 render：把虚拟树画到 canvas
function drawCanvas(vnode) {
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  drawNode(vnode);
}

function drawNode(node) {
  if (node.tag === 'rect') {
    const { x, y, width, height, fill } = node.props;
    ctx.fillStyle = fill;
    ctx.fillRect(x, y, width, height);
  } else if (node.tag === 'circle') {
    const { cx, cy, r, fill } = node.props;
    ctx.beginPath();
    ctx.arc(cx, cy, r, 0, Math.PI * 2);
    ctx.fillStyle = fill;
    ctx.fill();
  }
  if (node.children) {
    node.children.forEach(drawNode);
  }
}

// 使用
const App = {
  render() {
    return {
      tag: 'rect',
      props: { x: 10, y: 10, width: 100, height: 100, fill: 'red' },
      children: []
    };
  }
};

const app = renderer.createApp(App);
app.mount({ children: [] });
drawCanvas(app._container);
```

### 实际应用

1. **@vue/runtime-dom**：浏览器 DOM 渲染器（默认）。
2. **@vue/runtime-test**：测试用渲染器，记录所有操作，用于单元测试。
3. **Vue Native / Nativescript-Vue**：渲染到原生移动组件。
4. **自定义 Canvas/WebGL 渲染器**：游戏、数据可视化。
5. **SSR 渲染器**：渲染到字符串（`renderToString`）。

## 渲染器与响应式的结合

```js
function mountComponent(vnode, container, anchor) {
  const instance = vnode.component = createComponentInstance(vnode);

  setupComponent(instance); // 执行 setup

  // 创建渲染 effect
  instance.update = effect(() => {
    if (!instance.isMounted) {
      mountComponent(instance, container, anchor);
      instance.isMounted = true;
    } else {
      patch(instance.subTree, nextTree, container, anchor);
    }
  }, {
    scheduler: () => {
      queueJob(instance.update); // 异步批量更新
    }
  });
}
```

**关键：** 组件的 render 函数被 effect 包裹，render 时访问响应式数据会收集依赖。数据变化时，effect 的 scheduler 把更新任务入队，nextTick 时重新 render + patch。

## 总结

- **渲染器**：把 VNode 转成目标平台的真实结构（DOM/Canvas/原生组件）。
- **平台无关**：通过 `createRenderer(options)` 传入平台操作函数，核心逻辑复用。
- **patch**：根据 VNode 类型（元素/组件/Fragment/Text/Teleport/Suspense）分发处理。
- **自定义渲染器**：实现平台操作函数，可渲染到任意目标。
- **与响应式结合**：render 被 effect 包裹，数据变化触发重新 patch。

Vue3 的渲染器设计是"关注点分离"的典范：核心算法（diff/patch/组件管理）与平台操作（DOM API）解耦，让 Vue 能跨平台运行。
